		name = sik.2.a

			factor = 0.5

				factor = 10
				is_historical_focus_on = yes
			}
		}
		ROOT = {
			add_ai_strategy = {
				type = antagonize
				id = "TNG"
				value = 200
			}
			add_ai_strategy = {
				type = conquer
				id = "TNG"
				value = 500
			}
			if = {
				limit = {
					TNG = {
						is_guaranteed_by = ROOT
					}
				}
				diplomatic_relation = { country = TNG relation = guarantee active = no }
			}
			declare_war_on = {
					target = TNG
					type = annex_everything
				}
			}
		hidden_effect = {
			every_country = { 
				if = {
					limit = {
						exists = yes
						has_capitulated = no
						OR = {
							is_neighbor_of = TNG
							tag = TNG
							is_subject_of = CHI
							original_tag = CHI
							original_tag = PRC
							is_subject_of = PRC
						}
					}
					news_event = { 
						days = 1
						id = r56.news_event.20
					}
				}
			}
			if = {
				limit = {
					SOV = {
						has_war = no
						has_government = communism
					}
					SIK = {
						has_government = communism
					}
				}
				SOV = {
					add_ai_strategy = {
						type = support
						id = "SIK"
						value = 250
					}
					add_ai_strategy = {
						type = send_volunteers_desire
						id = "SIK"
						value = 500
					}

						limit = {
							SIK = {
								is_ai = yes
							}
						}
						add_ai_strategy = {
							type = alliance
							id = "SIK"
							value = -10 #offset so they don't add them to the comintern. 
						}
					}
				}
			}
		}
	}

			type = antagonize
			id = "TNG"
			value = 250
		}

		}

			factor = 0

			}
			modifier = {
				add = 0.4

					is_ai = no
				}
			}
			modifier = {
				add = 0.4
				has_war_with_major = yes
			}
		}
	}
