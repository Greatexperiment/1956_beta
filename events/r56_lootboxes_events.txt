					create_corps_commander = {
						name = "Beefo the Destroyer"

						skill = 3
						attack_skill = 4
						defense_skill = 4

					}
					set_country_flag = r56_lootbox_skeleton_claimed_flag
				}
				10 = {
					create_field_marshal = {
						name = "Jeffo the Leader of All"
						portrait_path = "gfx/leaders/r56_generic/r56_portrait_generic_skeleton_3.tga"
						skill = 5

		}
		
		else = {
			random_army_leader = {
				add_unit_leader_trait = skeletons_blessing
			}
		}
